33 // Visual C++ 2005 and later require the source files in UTF-8, and all strings
34 // to be encoded as wchar_t otherwise the strings will be converted into the
35 // local multibyte encoding and cause errors. To use a wchar_t as UTF-8, these
36 // strings then need to be convert back to UTF-8. This function is just a rough
37 // example of how to do this.
38 #include <Windows.h>
39 #define utf8(str) ConvertToUTF8(L##str)
40 string ConvertToUTF8(const wchar_t* pStr) {
41  static char szBuf[1024];
42  WideCharToMultiByte(CP_UTF8, 0, pStr, -1, szBuf, sizeof(szBuf), NULL, NULL);
43  return szBuf;
44 }

47 // Visual C++ 2003 and gcc will use the string literals as is, so the files
48 // should be saved as UTF-8. gcc requires the files to not have a UTF-8 BOM.
49 #define utf8(str) string(str)
50 #endif // if defined(_MSC_VER) && _MSC_VER > 1310
