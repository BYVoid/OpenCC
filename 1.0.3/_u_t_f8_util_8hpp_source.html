39  static size_t NextCharLengthNoException(const char* str) {
40  char ch = *str;
41  if ((ch & 0xF0) == 0xE0) {
42  return 3;
43  } else if ((ch & 0x80) == 0x00) {
44  return 1;
45  } else if ((ch & 0xE0) == 0xC0) {
46  return 2;
47  } else if ((ch & 0xF8) == 0xF0) {
48  return 4;
49  } else if ((ch & 0xFC) == 0xF8) {
50  return 5;
51  } else if ((ch & 0xFE) == 0xFC) {
52  return 6;
53  }
54  return 0;
55  }
