35class OPENCC_EXPORT Exception {
36public:
37 Exception() {}
38
39 virtual ~Exception() throw() {}
40
41 Exception(const std::string& _message) : message(_message) {}
42
43 virtual const char* what() const noexcept { return message.c_str(); }
44
45protected:
46 std::string message;
47};
