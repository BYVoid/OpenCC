49 static size_t NextCharLengthNoException(const char* str) {
50 const unsigned char ch = static_cast<unsigned char>(*str);
51 if ((ch & 0xF0) == 0xE0) {
52 return 3;
53 } else if ((ch & 0x80) == 0x00) {
54 return 1;
55 } else if ((ch & 0xE0) == 0xC0) {
56 return 2;
57 } else if ((ch & 0xF8) == 0xF0) {
58 return 4;
59 } else if ((ch & 0xFC) == 0xF8) {
60 return 5;
61 } else if ((ch & 0xFE) == 0xFC) {
62 return 6;
63 }
64 return 0;
65 }

81 static size_t PrevCharLength(const char* str) {
82 const char* candidate = str - 1;
83 size_t distance = 1;
84 while (distance < 6) {
85 const unsigned char ch = static_cast<unsigned char>(*candidate);
86 if ((ch & 0xC0) != 0x80) {
87 break;
88 }
89 candidate--;
90 distance++;
91 }
92
93 const size_t length = NextCharLengthNoException(candidate);
94 if (length == distance) {
95 return length;
96 }
97 throw InvalidUTF8(str);
98 }

120 static size_t Length(const char* str) {
121 size_t length = 0;
122 while (*str != '\0') {
123 const size_t charLen = NextCharLengthNoException(str);
124 if (charLen == 0) {
125 throw InvalidUTF8(str);
126 }
127 // Verify all continuation bytes are present before the null terminator.
128 // Use a while loop (not a for-with-return) to avoid complex control flow
129 // that triggers MSVC LTCG code-generator bugs.
130 size_t i = 1;
131 while (i < charLen && str[i] != '\0') {
132 ++i;
133 }
134 if (i < charLen) {
135 throw InvalidUTF8(str); // Truncated sequence: throw, don't silently skip
136 }
137 str += charLen;
138 ++length;
139 }
140 return length;
141 }

177 static bool NotShorterThan(const char* str, size_t byteLength) {

183 str++;

192 static std::string TruncateUTF8(const char* str, size_t maxByteLength) {
193 std::string wordTrunc;
194 if (NotShorterThan(str, maxByteLength)) {
195 size_t len = 0;
196 const char* pStr = str;
197 for (;;) {
198 const size_t charLength = NextCharLength(pStr);
199 if (len + charLength > maxByteLength) {
200 break;
201 }
202 pStr += charLength;
203 len += charLength;
204 }
205 wordTrunc = FromSubstr(str, len);
206 } else {
207 wordTrunc = str;
208 }
209 return wordTrunc;
210 }

215 static void ReplaceAll(std::string& str, const char* from, const char* to) {
216 std::string::size_type pos = 0;
217 std::string::size_type fromLen = strlen(from);
218 std::string::size_type toLen = strlen(to);
219 while ((pos = str.find(from, pos)) != std::string::npos) {
220 str.replace(pos, fromLen, to);
221 pos += toLen;
222 }
223 }

253 static void GetByteMap(const char* str, const size_t utf8Length,
254 std::vector<size_t>* byteMap) {
255 if (byteMap->size() < utf8Length) {
256 byteMap->resize(utf8Length);
257 }
258 const char* pstr = str;
259 for (size_t i = 0; i < utf8Length; i++) {
