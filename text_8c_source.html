37 int parse_entry(const char* buff, TextEntry* entry_i) {
38  size_t length;
39  const char* pbuff;
40 
41  /* 解析鍵 */
42  for (pbuff = buff; *pbuff != '\t' && *pbuff != '\0'; ++pbuff) {}
43 
44  if (*pbuff == '\0') {
45  return -1;
46  }
47  length = pbuff - buff;
48 
49  ucs4_t* ucs4_buff;
50  ucs4_buff = utf8_to_ucs4(buff, length);
51 
52  if (ucs4_buff == (ucs4_t*)-1) {
53  return -1;
54  }
55  entry_i->key = (ucs4_t*)malloc((length + 1) * sizeof(ucs4_t));
56  ucs4cpy(entry_i->key, ucs4_buff);
57  free(ucs4_buff);
58 
59  /* 解析值 */
60  size_t value_i, value_count = INITIAL_DICTIONARY_SIZE;
61  entry_i->value = (ucs4_t**)malloc(value_count * sizeof(ucs4_t*));
62 
63  for (value_i = 0; *pbuff != '\0' && *pbuff != '\n'; ++value_i) {
64  if (value_i >= value_count) {
65  value_count += value_count;
66  entry_i->value = (ucs4_t**)realloc(
67  entry_i->value,
68  value_count * sizeof(ucs4_t*)
69  );
70  }
71 
72  for (buff = ++pbuff;
73  *pbuff != ' ' && *pbuff != '\0' && *pbuff != '\n' && *pbuff != '\r';
74  ++pbuff) {}
75  length = pbuff - buff;
76  ucs4_buff = utf8_to_ucs4(buff, length);
77 
78  if (ucs4_buff == (ucs4_t*)-1) {
79  /* 發生錯誤 回退內存申請 */
80  ssize_t i;
81 
82  for (i = value_i - 1; i >= 0; --i) {
83  free(entry_i->value[i]);
84  }
85  free(entry_i->value);
86  free(entry_i->key);
87  return -1;
88  }
89 
90  entry_i->value[value_i] = (ucs4_t*)malloc((length + 1) * sizeof(ucs4_t));
91  ucs4cpy(entry_i->value[value_i], ucs4_buff);
92  free(ucs4_buff);
93  }
94 
95  entry_i->value = (ucs4_t**)realloc(
96  entry_i->value,
97  value_count * sizeof(ucs4_t*)
98  );
99  entry_i->value[value_i] = NULL;
100 
101  return 0;
102 }

104 Dict* dict_text_new(const char* filename) {
105  TextDict* text_dictionary;
106 
107  text_dictionary = (TextDict*)malloc(sizeof(TextDict));
108  text_dictionary->entry_count = INITIAL_DICTIONARY_SIZE;
109  text_dictionary->max_length = 0;
110  text_dictionary->lexicon = (TextEntry*)malloc(
111  sizeof(TextEntry) * text_dictionary->entry_count);
112  text_dictionary->word_buff = NULL;
113 
114  static char buff[ENTRY_BUFF_SIZE];
115 
116  FILE* fp = fopen(filename, "r");
117 
118  if (fp == NULL) {
119  dict_text_delete((Dict*)text_dictionary);
120  return (Dict*)-1;
121  }
122  skip_utf8_bom(fp);
123 
124  size_t i = 0;
125 
126  while (fgets(buff, ENTRY_BUFF_SIZE, fp)) {
127  if (i >= text_dictionary->entry_count) {
128  text_dictionary->entry_count += text_dictionary->entry_count;
129  text_dictionary->lexicon = (TextEntry*)realloc(
130  text_dictionary->lexicon,
131  sizeof(TextEntry) * text_dictionary->entry_count
132  );
133  }
134 
135  if (parse_entry(buff, text_dictionary->lexicon + i) == -1) {
136  text_dictionary->entry_count = i;
137  dict_text_delete((Dict*)text_dictionary);
138  return (Dict*)-1;
139  }
140 
141  size_t length = ucs4len(text_dictionary->lexicon[i].key);
142 
143  if (length > text_dictionary->max_length) {
144  text_dictionary->max_length = length;
145  }
146 
147  i++;
148  }
149 
150  fclose(fp);
151 
152  text_dictionary->entry_count = i;
153  text_dictionary->lexicon = (TextEntry*)realloc(
154  text_dictionary->lexicon,
155  sizeof(TextEntry) * text_dictionary->entry_count
156  );
157  text_dictionary->word_buff = (ucs4_t*)
158  malloc(sizeof(ucs4_t) *
159  (text_dictionary->max_length + 1));
160 
161  qsort(text_dictionary->lexicon,
162  text_dictionary->entry_count,
163  sizeof(text_dictionary->lexicon[0]),
164  qsort_entry_cmp
165  );
166 
167  return (Dict*)text_dictionary;
168 }

191 const ucs4_t* const* dict_text_match_longest(Dict* dict,
192  const ucs4_t* word,
193  size_t maxlen,
194  size_t* match_length) {
195  TextDict* text_dictionary = (TextDict*)dict;
196 
197  if (text_dictionary->entry_count == 0) {
198  return NULL;
199  }
200 
201  if (maxlen == 0) {
202  maxlen = ucs4len(word);
203  }
204  size_t len = text_dictionary->max_length;
205 
206  if (maxlen < len) {
207  len = maxlen;
208  }
209 
210  ucs4ncpy(text_dictionary->word_buff, word, len);
211  text_dictionary->word_buff[len] = L'\0';
212 
213  TextEntry buff;
214  buff.key = text_dictionary->word_buff;
215 
216  for (; len > 0; len--) {
217  text_dictionary->word_buff[len] = L'\0';
218  TextEntry* brs = (TextEntry*)bsearch(
219  &buff,
220  text_dictionary->lexicon,
221  text_dictionary->entry_count,
222  sizeof(text_dictionary->lexicon[0]),
223  qsort_entry_cmp
224  );
225 
226  if (brs != NULL) {
227  if (match_length != NULL) {
228  *match_length = len;
229  }
230  return (const ucs4_t* const*)brs->value;
231  }
232  }
233 
234  if (match_length != NULL) {
235  *match_length = 0;
236  }
237  return NULL;
238 }
